10 //
11 // SymbolTable<DAT> describes a symbol table mapping symbols of
12 // type `SYM' to data of type `DAT *'. It is implemented as a
13 // list of lists of `SymtabEntry<SYM,DAT> *'. The inner list is
14 // a scope, a mapping from symbols to data, and the outer list is
15 // a list of scopes.
16 //
17 // `tbl' points to the current top scope.
18 //
19 // `enterscope' makes the table point to a new scope whose parent
20 // is the scope it pointed to previously.
21 //
22 // `exitscope' makes the table point to the parent scope of the
23 // current scope. Note that the old child scope is not
24 // deallocated. One may save the state of a symbol table
25 // at a given point by copying it with `operator ='
26 //
27 // `put(s,i)' adds a symbol table entry to the current scope of
28 // the symbol table mapping symbol `s' to data `d'. The old
29 // top scope isn't modified; a new scope is created whose
30 // entry list is the new entry followed by the old entry list,
31 // and whose tail is the old top scope's parent. The table
32 // is made to point to this new scope.
33 //
34 // `lookup(s)' looks for the symbol `s', starting at the top scope
35 // and proceeding down the list of scopes until either an
36 // entry is found whose `get_id()' equals `s', or the end of
37 // the root scope is reached. It returns the data item
38 // associated with the entry, or NULL if no such entry exists.
39 //
40 //
41 // `probe(s)' checks the top scope for an entry whose `get_id()'
42 // equals `s', and returns the entry's `get_info()' if
43 // found, and NULL otherwise.
44 //
45 // `dump()' prints the symbols in the symbol table.
46 //
47 
48 template <class DAT>
49 class SymbolTable
50 {
51  using Scope = std::unordered_map<string, DAT>;
52  using ScopeList = std::list<Scope>;
53 private:
54  ScopeList *scopes;
55 public:
56  SymbolTable(){ scopes = new std::list<Scope>();} // create a new symbol table
57 
58  // Create pointer to current symbol table.
59  SymbolTable &operator =(const SymbolTable &s) { scopes = s.scopes; return *this; }
60 
61  void fatal_error(string msg)
62  {
63  cerr << msg << "\n";
64  exit(1);
65  }
66 
67  // Enter a new scope. A symbol table is organized as a list of
68  // lists. The head of the list is the innermost scope, the tail
69  // holds the outer scopes. A scope must be entered before anything
70  // can be added to the table.
71 
72  void enterscope()
73  {
74  scopes->push_front(std::unordered_map<string, DAT>());
75  }
76 
77  // Pop the first scope off of the symbol table.
78  void exitscope()
79  {
80  // It is an error to exit a scope that doesn't exist.
81  if (scopes->empty()) {
82  fatal_error("exitscope: Can't remove scope from an empty symbol table.");
83  }
84  scopes->pop_front();
85  }
86 
87  // put an item to the symbol table.
88  void put(string s, DAT i)
89  {
90  // There must be at least one scope to add a symbol.
91  if (scopes->empty()) fatal_error("addid: Can't add a symbol without a scope.");
92 
93  Scope& scope = scopes->front();
94  scope[s] = i;
95  }
96 
97  // Lookup an item through all scopes of the symbol table. If found
98  // it returns the associated information field, if not it returns
99  // NULL.
100 
101  DAT lookup(string s)
102  {
103  for (auto iter = scopes->cbegin(); iter != scopes->cend(); iter++) {
104  auto search = iter->find(s);
105  if (search != iter->end()) {
106  return search->second;
107  }
108  }
109  return NULL;
110  }
111 
112  // probe the symbol table. Check the top scope (only) for the item
113  // 's'. If found, return the information field. If not return NULL.
114  DAT probe(string s)
115  {
116  if (scopes->empty()) {
117  fatal_error("probe: No scope in symbol table.");
118  }
119  Scope& current = scopes->front();
120  auto search = current.find(s);
121  if (search != current.end()) {
122  return search->second;
123  }
124  return(NULL);
125  }
126 
127  // Prints out the contents of the symbol table
128  void dump()
129  {
130  for (auto pscope = scopes->cbegin(); pscope != scopes->cend(); pscope++) {
131  cerr << "\nScope: ";
132  for (auto & entry : *pscope)
133  cerr << ' ' << entry.first;
134  }
135  }
136 
137 };
138 #endif //SMALLC_SYMTAB_H
